class ChannelButton : public QPushButton {
    Q_OBJECT
public:
    ChannelButton(QColor baseColor, QWidget* parent = nullptr)
        : QPushButton(parent), _baseColor(baseColor) {
        setCheckable(true);
        setFixedSize(22, 22);
        setFlat(true);
    }
protected:
    void paintEvent(QPaintEvent* event) override {
        QPainter painter(this);
        painter.setRenderHint(QPainter::Antialiasing);
        QColor fillColor = isEnabled() ? _baseColor : QColor(60, 60, 60);
        if (isChecked() && isEnabled()) fillColor = fillColor.lighter(130);
        painter.fillRect(rect(), fillColor);
        painter.setPen(QPen(QColor(30, 30, 30), 1));
        painter.drawRect(rect().adjusted(0, 0, -1, -1));

            painter.setPen(QPen(Qt::white, 2));
            painter.drawLine(4, 4, 17, 17);
            painter.drawLine(17, 4, 4, 17);
        }
    }
private:
    QColor _baseColor;
};
