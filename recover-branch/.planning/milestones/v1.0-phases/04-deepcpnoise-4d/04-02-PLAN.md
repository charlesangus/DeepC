// Pattern: for each axis compute w0=floor-1, w1=floor, w2=floor+1, w3=floor+2

FN_DECIMAL FastNoise::SingleCubicFractalFBM(FN_DECIMAL x, FN_DECIMAL y, FN_DECIMAL z, FN_DECIMAL w) const
{

    while (++i < m_octaves) { x *= m_lacunarity; y *= m_lacunarity; z *= m_lacunarity; w *= m_lacunarity; amp *= m_gain; sum += SingleCubic(m_perm[i], x, y, z, w) * amp; }

FN_DECIMAL FastNoise::SingleCubicFractalBillow(FN_DECIMAL x, FN_DECIMAL y, FN_DECIMAL z, FN_DECIMAL w) const

{
    int xr = FastRound(x);
    int yr = FastRound(y);
    int zr = FastRound(z);
    int wr = FastRound(w);

    FN_DECIMAL distance = FN_DECIMAL(999999);
    int xcell = 0, ycell = 0, zcell = 0, wcell = 0;

    {
        for (int zi = zr - 1; zi <= zr + 1; zi++)
        {
